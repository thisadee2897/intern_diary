```python
def spin_words(sentence):
    words = [i for i in sentence.split(" ")]
    out_words =[]
    new_sentence  = ' '
    for word in words:
        langht = len(word) 
        if 4 < langht:
            new_word = word[::-1]
            out_words.append(new_word)
        else:
            new_word = word
            out_words.append(new_word)
    return(new_sentence.join(out_words))
```
